\begin{align*} \frac{2}{n+1} \sum_{m=1}^{n/2} \left\lceil \frac{2m}{c} \right\rceil &\leq \frac{2}{n+1} \sum_{m=1}^{n/2} \left( \frac{2m}{c} + 1\right) = \frac{2}{n+1} \left( \frac{2}{c} \frac{(n/2) (n/2+1)}{2} + \frac{n}{2} \right)\\ &= \frac{2}{n+1} \left( \frac{n(n+2)}{4c} + \frac{n}{2} \right) \leq \frac{n}{2c} + 1 \end{align*}
