\begin{align*} \E[X^m] &= C \int_{x=k_{\min}}^{n} x^m \cdot x^{-\alpha} dx = C \int_{x=k_{\min}}^{n} x^{-\alpha+m} dx\\ &= \frac{C}{-\alpha +m+1} x^{-\alpha+n+1} \Big|_{k_{\min}}^n\\ &= \frac{C}{-\alpha +m+1} \left( \frac{1}{n^{\alpha-m-1}} - \frac{1}{k_{\min}^{\alpha-m-1}} \right) \end{align*}
