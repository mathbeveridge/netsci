(n-k)!} p^{k-1} (1-p)^{n-k}\\ & = np \sum_{j=0}^{n-1} \frac{(n-1)!}{j! (n-1-j)!} p^{j} (1-p)^{n-1-j}\\ &= np \sum_{j=0}^{n-1} {n-1 \choose j} p^{j} (1-p)^{n-1-j}\\ & = np (p + (1-p))^{n-1} \, = \, np. \end{align*}
